116     /**
117      * Inserts the number of specified rows with each row containing a column,
118      * {@link TestConstants#COLUMN1}, with the value provided and another column,
119      * {@link TestConstants#COLUMN2}, with an arbitrary value
120      *
121      * @param proxy          The proxy to use for invoking a scan, not null
122      * @param rowCount       The valid number of rows expected to be returned from the scan
123      * @param keyColumnValue The value stored in the first column, which is intended to be indexed
124      * @param uuid           The unique identifier to associate with each row inserted, not null
125      */
126     public static void insertData(final HandlerProxy proxy, final int rowCount,
127                                   final long keyColumnValue, final UUID uuid) {
128         checkNotNull(proxy);
129         verifyRowCount(rowCount);
130         checkNotNull(uuid);
131 
132         final Map<String, ByteBuffer> map = Maps.newHashMap();
133         map.put(TestConstants.COLUMN1, encodeValue(keyColumnValue));
134 
135         for (int x = 0; x < rowCount; x++) {
136             map.put(TestConstants.COLUMN2, encodeValue(x));
137             final Row row = new Row(map, uuid);
138             proxy.insertRow(row.serialize());
139         }
140 
141         proxy.flush();
142     }

202     private static void assertDifferentRows(final HandlerProxy proxy, final int rowCount) {
203         byte[] previous = null;
204 
205         for (int x = 0; x < rowCount; x++) {
206             final byte[] current = proxy.getNextRow();
207             assertNotNull(current);
208             assertThat(current, not(equalTo(previous)));
209             previous = current;
210         }
211 
212         assertNull(proxy.getNextRow());
213     }

219     /**
220      * Check that the table open on the proxy has the expected number of data rows
221      * and index rows on each index (checks both ascending and descending
222      * directions).  Note: this could be very slow for big tables.
223      * @param proxy HandlerProxy with table already open
224      * @param schema TableSchema of open table
225      * @param expectedRowCount Expected number of rows
226      */
227     public static void assertRowCount(final HandlerProxy proxy,
228                                       final TableSchema schema,
229                                       final long expectedRowCount) {
230         checkState(proxy.getTableName() != null, "Proxy must have an open table.");
231         checkNotNull(schema);
232         verifyRowCount(expectedRowCount);
233 
234         proxy.startTableScan();
235         assertResultCount(proxy, expectedRowCount);
236         proxy.endScan();
237 
238         QueryKey queryKey;
239         for(IndexSchema indexSchema : schema.getIndices()) {
240             queryKey = new QueryKey(indexSchema.getIndexName(),
241                     QueryType.INDEX_FIRST, ImmutableMap.<String, ByteBuffer>of());
242             proxy.startIndexScan(queryKey.serialize());
243             assertResultCount(proxy, expectedRowCount);
244             proxy.endScan();
245 
246             queryKey = new QueryKey(indexSchema.getIndexName(),
247                     QueryType.INDEX_LAST, ImmutableMap.<String, ByteBuffer>of());
248             proxy.startIndexScan(queryKey.serialize());
249             assertResultCount(proxy, expectedRowCount);
250             proxy.endScan();
251         }
252     }

255      * Checks that the open scan on the proxy has the expected number of results.
256      * @param proxy
257      * @param expectedResultCount
258      */
259     private static void assertResultCount(final HandlerProxy proxy,
260                                           final long expectedResultCount) {
261         long actualResultCount = 0;
262         while(proxy.getNextRow() != null) {
263             actualResultCount++;
264         }
265         assertEquals(expectedResultCount, actualResultCount);

268     /**
269      * Start concurrency number of actions concurrently and at the same time.  This
270      * method ensures that the actions are started at the same time, and they are
271      * not being blocked in a thread pool.  Obviously, this uses concurrency
272      * number of threads simultaneously so use caution as thread starvation
273      * deadlock can occur with high concurrency levels.
274      *
275      * @param concurrency Number of handler proxies to run the action against, concurrently
276      * @param setup Action to be run on each proxy before starting concurrent actions
277      * @param action Action to be run concurrently and in sync with other proxies
278      * @param cleanup Action to be run to cleanup proxy
279      * @param factory HandlerProxyFactory to supply proxies to be run concurrently
280      * @throws InterruptedException
281      */
282     public static void startProxyActionConcurrently(int concurrency,
283                                                     final ProxyRunnable setup,
284                                                     final ProxyRunnable action,
285                                                     final ProxyRunnable cleanup,
286                                                     final HandlerProxyFactory factory)
287             throws InterruptedException{
288         final ExecutorService executor = Executors.newCachedThreadPool();
289         final CountDownLatch ready = new CountDownLatch(concurrency);
290         final CountDownLatch start = new CountDownLatch(1);
291         final CountDownLatch done = new CountDownLatch(concurrency);
292 
293         for (int i = 0; i < concurrency; i++) {
294             executor.execute(new Runnable() {
295                 @Override
296                 public void run() {
297                     HandlerProxy proxy = factory.createHandlerProxy();
298                     setup.run(proxy);
299                     ready.countDown();
300                     try {
301                         start.await();
302                         action.run(proxy);
303                     } catch (InterruptedException e) {
304                         System.out.println("Interuppted");
305                         Thread.currentThread().interrupt();
306                     }
307                     cleanup.run(proxy);
308                     done.countDown();
309                 }
310             });
311         }
312         ready.await();
313         start.countDown();
314         done.await();
315     }
