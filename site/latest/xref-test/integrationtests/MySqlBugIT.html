47  /**
48   * Integration tests for the Java side that come from MySQL integration tests.
49   * If a MySQL integration test fails and it's not a C++ issue it should appear in here
50   * in the format:
51   * SQL query that failed
52   * What was expected
53   * What happened
54   * Test reproducing the failure
55   */
56  public class MySqlBugIT extends HoneycombIntegrationTest {
57      private static final int ROW_COUNT = 1;
58      private static final int INDEX_COL_VALUE = 7;

60      /*
61      create table t1 (c1 int, c2 int, unique(c1,c2));
62      insert into t1 values (1,1),(1,1);
63      Expected failure
64       */
65      @Test
66      public void testRowInsertNotExcludingFieldsFromIndexKey() {
67          ITUtils.insertData(proxy, 1, 1);
68          Row row = ITUtils.createRow(1);
69          assertTrue(proxy.indexContainsDuplicate(TestConstants.INDEX1, row.serialize()));
70      }

72      /*
73      DROP TABLE if exists t1;
74      CREATE TABLE t1(c1 BIGINT SIGNED NULL,c2 BIGINT SIGNED NULL , INDEX(c1, c2));
75      INSERT INTO t1 VALUES
76      (4611686018427387903, NULL),
77      (4611686018427387903, NULL);
78  
79      SELECT * FROM t1 WHERE c1 = 4611686018427387903 AND c2 is null ORDER BY c1 ASC;
80      Expected two rows
81      Actual was zero rows
82       */
83      @Test
84      public void testInsertBuildsIndexCorrectlyOnNullColumns() {
85          final Map<String, ByteBuffer> map = Maps.newHashMap();
86          map.put(TestConstants.COLUMN1, ITUtils.encodeValue(INDEX_COL_VALUE));
87  
88          final Row row = new Row(map, UUID.randomUUID());
89          proxy.insertRow(row.serialize());
90          proxy.flush();
91  
92          HashMap<String, ByteBuffer> keys = Maps.newHashMap();
93          keys.put(TestConstants.COLUMN1, ITUtils.encodeValue(INDEX_COL_VALUE));
94          keys.put(TestConstants.COLUMN2, null);
95  
96          final QueryKey key = new QueryKey(TestConstants.INDEX2, QueryType.EXACT_KEY, keys);
97  
98          ITUtils.assertReceivingDifferentRows(proxy, key, ROW_COUNT);
99      }

101     /*
102     DROP TABLE IF EXISTS t1;
103     CREATE TABLE t1(c1 INT, c2 INT, INDEX(c1));
104     INSERT INTO t1 VALUES(NULL, 1);
105     UPDATE t1 SET c2=2 WHERE c1 IS NULL;
106     Index scan: Null, 1
107     Table scan: Null, 2
108      */
109     @Test
110     public void testUpdateNotChangingIndicesWhenUpdatedColumnNotInIndex() {
111         Map<String, ByteBuffer> values = Maps.newHashMap();
112         values.put(TestConstants.COLUMN2, ITUtils.encodeValue(1));
113         Row row = new Row(values, UUID.randomUUID());
114         proxy.insertRow(row.serialize());
115         proxy.flush();
116 
117         proxy.startTableScan();
118         byte[] nextRow = proxy.getNextRow();
119         row = Row.deserialize(nextRow);
120         proxy.endScan();
121         row.getRecords().put(TestConstants.COLUMN2, ITUtils.encodeValue(2)); // update t1 set c2=2 where c1 is null
122         proxy.updateRow(nextRow, row.serialize());
123 
124         Map<String, ByteBuffer> searchMap = Maps.newHashMap();
125         searchMap.put(TestConstants.COLUMN1, null);
126         QueryKey key = new QueryKey(TestConstants.INDEX1, QueryType.EXACT_KEY, searchMap);
127         proxy.startIndexScan(key.serialize());
128         Row result = Row.deserialize(proxy.getNextRow());
129         assertEquals(result.getRecords().get(TestConstants.COLUMN2).getLong(), ITUtils.encodeValue(2).getLong());
130         proxy.endScan();
131     }

136     insert into t1 values (null),(null);

140     @Test

143         Map<String, ByteBuffer> values = Maps.newHashMap();

145         Row row = new Row(values, UUID.randomUUID());

147     }

150     /**
151      * Provides the {@link com.nearinfinity.honeycomb.mysql.schema.TableSchema} to use for a test case
152      *
153      * @return The schema used for testing
154      */
155     @Override
156     protected TableSchema getTableSchema() {
157         final List<ColumnSchema> columns = Lists.newArrayList();
158         final List<IndexSchema> indices = Lists.newArrayList();
159 
160         // Add nullable, non-autoincrementing columns
161         columns.add(ColumnSchema.builder(TestConstants.COLUMN1, ColumnType.LONG).build());
162         columns.add(ColumnSchema.builder(TestConstants.COLUMN2, ColumnType.LONG).build());
163 
164         // Add non-unique index on one column
165         indices.add(new IndexSchema(TestConstants.INDEX1, Lists.newArrayList(TestConstants.COLUMN1), false));
166 
167         // Add non-unique compound index on (c1, c2)
168         indices.add(new IndexSchema(TestConstants.INDEX2, Lists.newArrayList(TestConstants.COLUMN1, TestConstants.COLUMN2), false));
169 
170         // Add unique index on one column
171         indices.add(new IndexSchema(TestConstants.INDEX3, Lists.newArrayList(TestConstants.COLUMN1), true));
172 
173         return new TableSchema(columns, indices);
174     }
