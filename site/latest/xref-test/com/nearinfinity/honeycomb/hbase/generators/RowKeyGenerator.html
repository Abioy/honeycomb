58  public class RowKeyGenerator implements Generator<RowKey> {
59      private static final Random RAND = new Random();
60      private static final TablesRowKey tablesRow = new TablesRowKey();
61      private static final RowsRowKey rowsRow = new RowsRowKey();
62      private static final AutoIncRowKey autoIncRow = new AutoIncRowKey();
63      private static final SchemaRowKey schemaRow = new SchemaRowKey();
64      private static final Generator<Long> randIdGen = CombinedGenerators.uniqueValues(
65              PrimitiveGenerators.longs(0, 1024));
66      private static final Generator<SortOrder> randSortOrder = PrimitiveGenerators.enumValues(SortOrder.class);
67      private static final Generator<UUID> uuidGen = new UUIDGenerator();
68      private static final Generator<TableSchema> tableSchemaGen = new TableSchemaGenerator(1);
69      private final FrequencyGenerator<RowKey> rowKeyGen;
70  
71      public RowKeyGenerator() {
72          // The duplicated generator types are testing the sorts of the different
73          // parts of the row key.  E.G. There are two ColumnSchema generators.
74          // The first tests sorting on tableId, the second holds tableId constant
75          // and tests sorting on columnId.
76  
77          rowKeyGen = new DefaultFrequencyGenerator<RowKey>(new PrefixRowGenerator(), 1);
78          rowKeyGen.add(new TableIDRowGenerator(), 1);
79          rowKeyGen.add(new DataRowKeyGenerator(randIdGen), 3);
80          rowKeyGen.add(new DataRowKeyGenerator(fixedLong()), 3);
81          rowKeyGen.add(new IndexRowKeyGenerator(randIdGen, randIdGen, tableSchemaGen.next(), randSortOrder), 4);
82          rowKeyGen.add(new IndexRowKeyGenerator(fixedLong(), randIdGen, tableSchemaGen.next(), randSortOrder), 4);
83          rowKeyGen.add(new IndexRowKeyGenerator(fixedLong(), fixedLong(), tableSchemaGen.next(), randSortOrder), 8);
84  
85      }
86  
87      private Generator<Long> fixedLong() {
88          return PrimitiveGenerators.fixedValues(randIdGen.next());
89      }
90  
91      public static IndexRowKeyGenerator getAscIndexRowKeyGenerator(TableSchema schema) {
92          return new IndexRowKeyGenerator(
93                  PrimitiveGenerators.fixedValues(randIdGen.next()),
94                  PrimitiveGenerators.fixedValues(randIdGen.next()),
95                  schema,
96                  PrimitiveGenerators.fixedValues(SortOrder.Ascending)
97          );
98      }

100     public static IndexRowKeyGenerator getDescIndexRowKeyGenerator(TableSchema schema) {
101         return new IndexRowKeyGenerator(
102                 PrimitiveGenerators.fixedValues(randIdGen.next()),
103                 PrimitiveGenerators.fixedValues(randIdGen.next()),
104                 schema,
105                 PrimitiveGenerators.fixedValues(SortOrder.Descending)
106         );
107     }

109     @Override
110     public RowKey next() {
111         return rowKeyGen.next();

114     private class PrefixRowGenerator implements Generator<RowKey> {
115         @Override
116         public RowKey next() {
117             switch (RAND.nextInt(4)) {
118                 case 0:
119                     return tablesRow;
120                 case 1:
121                     return rowsRow;
122                 case 2:
123                     return autoIncRow;
124                 case 3:
125                     return schemaRow;
126                 default:
127                     throw new RuntimeException("Should never reach me");
128             }
129         }
130     }
131 
132     private class TableIDRowGenerator implements Generator<RowKey> {
133         @Override
134         public RowKey next() {
135             if (RAND.nextBoolean()) {
136                 return new ColumnsRowKey(randIdGen.next());
137             }
138 
139             return new IndicesRowKey(randIdGen.next());
140         }
141     }

143     private class DataRowKeyGenerator implements Generator<RowKey> {
144         Generator<Long> tableIdGen;
145 
146         public DataRowKeyGenerator(Generator<Long> generator) {
147             tableIdGen = generator;
148         }
149 
150         @Override
151         public RowKey next() {
152             return new DataRowKey(tableIdGen.next(), uuidGen.next());
153         }
154     }

156     public static class IndexRowKeyGenerator implements Generator<RowKey> {
157         private final Generator<Long> tableIds;
158         private final Generator<Long> indexIds;
159         private final TableSchema tableSchema;
160         private final Generator<IndexSchema> indexSchemas;
161         private final Generator<Row> rows;
162         private final Generator<QueryKey> queryKeys;
163         private final Generator<SortOrder> order;
164 
165         public IndexRowKeyGenerator(
166                 Generator<Long> tableIds,
167                 Generator<Long> indexIds,
168                 TableSchema tableSchema,
169                 Generator<SortOrder> order) {
170             this.tableIds = tableIds;
171             this.indexIds = indexIds;
172             this.tableSchema = tableSchema;
173             Collection<IndexSchema> indices = this.tableSchema.getIndices();
174             checkState(indices.size() > 0, "Generated table schema must have an index.");
175             indexSchemas = PrimitiveGenerators.fixedValues(indices);
176             rows = new RowGenerator(this.tableSchema);
177             queryKeys = new QueryKeyGenerator(this.tableSchema);
178             this.order = order;
179         }
180 
181         @Override
182         public RowKey next() {
183             IndexRowKeyBuilder builder;
184             Row row = rows.next();
185             builder = IndexRowKeyBuilder
186                     .newBuilder(tableIds.next(), indexIds.next())
187                     .withRow(row,
188                             indexSchemas.next().getIndexName(),
189                             tableSchema)
190                     .withUUID(row.getUUID());
191 
192             return builder.withSortOrder(order.next()).build();
193         }
194 
195         public Pair<IndexRowKey, QueryKey> nextWithQueryKey() {
196             QueryKey queryKey = queryKeys.next();
197             IndexRowKey rowKey = IndexRowKeyBuilder
198                     .newBuilder(tableIds.next(), indexIds.next())
199                     .withQueryKey(queryKey, tableSchema)
200                     .withSortOrder(order.next())
201                     .build();
202             return new Pair<IndexRowKey, QueryKey>(rowKey, queryKey);
203         }
204     }
